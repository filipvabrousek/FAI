```sql
CREATE FUNCTION fnbal_invoice()  
RETURNS MONEY  
BEGIN  
RETURN(SELECT SUM(invoicetotal-paymenttotal-creadittotal)  
        FROM dbo.mcninvoices  
        WHERE invoicetotal-paymenttotal-creadittotal > 0 )  
END
```
